from scipy.optimize import linear_sum_assignment
import networkx as nx

def optimize_routes(morning_routes, evening_routes):
    # Build cost matrix from road distances
    cost_matrix = build_cost_matrix(
        morning_routes,
        evening_routes,
        kakao_api_client
    )

    # Hungarian Algorithm for optimal matching
    row_ind, col_ind = linear_sum_assignment(cost_matrix)

    # Validate constraints
    valid_pairs = []
    for i, j in zip(row_ind, col_ind):
        if validate_time_constraint(i, j, max_time=50):
            valid_pairs.append((morning_routes[i], evening_routes[j]))

    return valid_pairs
